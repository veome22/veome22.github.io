def test_accuracy(predicted_greens, data_greens):
    results = np.array(predicted_greens, dtype=int) == np.array(data_greens, dtype=int)
    bins=[0.0, 0.5, 1.0]
    hist, binss = np.histogram(results, bins=bins)
    wrong = hist[0]
    right = hist[1]
    
    return right, wrong

def sim_return(predicted_greens, daily_change):
    predicted_greens = np.array(predicted_greens, dtype=bool)
    return np.sum(daily_change[predicted_greens])
In [207]:
# simulate the return, but with some actual principal amount
def sim_realistic_return(predicted_greens, principal, daily_change_percent):
    amount = principal
    percents = daily_change_percent[np.array(predicted_greens, dtype=bool)]
    
    for i in range(len(percents)):
        amount = amount + (amount*percents[i])

    return amount

from statsmodels.tsa.stattools import adfuller
import numpy as np
def check_stationarity(ts):
    dftest = adfuller(ts)
    adf = dftest[0]
    pvalue = dftest[1]
    critical_value = dftest[4]['5%']
    if (pvalue < 0.05) and (adf < critical_value):
        print('The series is stationary')
    else:
        print('The series is NOT stationary')
